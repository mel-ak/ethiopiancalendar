Documentation

The Ethiopian Calendar Go Library provides utilities for the Ethiopian (Ge'ez) calendar, a solar calendar with 13 months: 12 months of 30 days each, and Pagumē (13th) with 5 or 6 days in leap years. It lags the Gregorian calendar by ~7-8 years due to differences in the Annunciation calculation.

  • Date Representation: EtDate struct with validation.
  • Leap Years: Determined by year % 4 == 3.
  • Conversions: Bidirectional Ethiopian ↔ Gregorian using Julian Day Number (JDN).
  • Formatting: Custom layouts with Amharic month names.
  • Arithmetic: Add/subtract days, months, years with normalization.
  • No Dependencies: Pure Go implementation.

Key Functions
  • IsLeap(year int) bool: Check if a year is a leap year.
  • DaysInMonth(year, month int) int: Get days in a month.
  • (EtDate).Validate() error: Validate an Ethiopian date.
  • (EtDate).ToGregorian() (int, int, int, error): Convert to Gregorian.
  • FromGregorian(year, month, day int) (EtDate, error): Convert from Gregorian.
  • (EtDate).Format(layout string) string: Format date with layout.
  • (EtDate).AddDays(days int) (EtDate, error): Add/subtract days.
  • (EtDate).AddMonths(months int) EtDate: Add/subtract months.
  • (EtDate).AddYears(years int) EtDate: Add/subtract years.

Date Representation and Validation
package main

import (
    "fmt"
    "github.com/mel-ak/ethiopiancalendar"
)

func main() {
    date := ethiopiancalendar.EtDate{Year: 2016, Month: 1, Day: 1}
    if err := date.Validate(); err != nil {
        fmt.Println("Invalid date:", err)
    } else {
        fmt.Println("Valid date:", date.Year, date.Month, date.Day)
    }
    // Output: Valid date: 2016 1 1
}
Leap Year Check
package main

import (
    "fmt"
    "github.com/mel-ak/ethiopiancalendar"
)

func main() {
    year := 2015 // 2015 EC is a leap year
    isLeap := ethiopiancalendar.IsLeap(year)
    days := ethiopiancalendar.DaysInMonth(year, 13)
    fmt.Printf("Year %d is leap: %v, Days in Pagumē: %d\n", year, isLeap, days)
    // Output: Year 2015 is leap: true, Days in Pagumē: 6
}
Ethiopian to Gregorian Conversion
package main

import (
    "fmt"
    "github.com/mel-ak/ethiopiancalendar"
)

func main() {
    date := ethiopiancalendar.EtDate{Year: 2016, Month: 1, Day: 1}
    y, m, d, err := date.ToGregorian()
    if err != nil {
        fmt.Println("Error:", err)
    } else {
        fmt.Printf("Gregorian: %d-%02d-%02d\n", y, m, d)
    }
    // Output: Gregorian: 2023-09-12
}
Gregorian to Ethiopian Conversion
package main

import (
    "fmt"
    "github.com/mel-ak/ethiopiancalendar"
)

func main() {
    date, err := ethiopiancalendar.FromGregorian(2023, 9, 12)
    if err != nil {
        fmt.Println("Error:", err)
    } else {
        fmt.Printf("Ethiopian: %d/%d/%d\n", date.Year, date.Month, date.Day)
    }
    // Output: Ethiopian: 2016/1/1
}

Arithmetic: Add/Subtract Days
package main

import (
    "fmt"
    "github.com/mel-ak/ethiopiancalendar"
)

func main() {
    date := ethiopiancalendar.EtDate{Year: 2016, Month: 1, Day: 1}
    newDate, err := date.AddDays(40)
    if err != nil {
        fmt.Println("Error:", err)
    } else {
        fmt.Printf("After adding 40 days: %d/%d/%d\n", newDate.Year, newDate.Month, newDate.Day)
    }
    // Output: After adding 40 days: 2016/2/11
}
